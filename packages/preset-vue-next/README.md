# @fanion/preset-vue-next

> Vue Next preset for Fanion

## Usage

```javascript
const fanion = require("@fanion/webpack");

module.exports = fanion({
  dashboard: {
    use: require("@fanion/preset-vue-next"),
  },
});
```
